<script>
    function revealSurprise() {
        const music = document.getElementById('valentine-music');
        const message = document.getElementById('secret-message');
        const btn = document.querySelector('.surprise-btn');

        // Play Music
        music.play().catch(e => console.log("Audio play blocked by browser."));
        
        // Show Message & Hide Button
        message.style.display = 'block';
        btn.style.display = 'none';
        
        // Immediate heart explosion!
        for(let i=0; i<40; i++) {
            setTimeout(createHeart, i * 40);
        }
    }

    function createHeart() {
        const heart = document.createElement('div');
        heart.classList.add('floating-heart');
        heart.innerHTML = '‚ù§Ô∏è';
        heart.style.left = Math.random() * 100 + 'vw';
        heart.style.bottom = "-5vh";
        heart.style.fontSize = (Math.random() * 20 + 10) + "px";
        heart.style.animationDuration = (Math.random() * 2 + 3) + 's';
        document.body.appendChild(heart);
        
        setTimeout(() => heart.remove(), 4000);
    }

    // Background drifting hearts
    setInterval(createHeart, 800);
